#8) Find the average Prevailing Wage for each Job for each Year (take part time and full time separate)

select job_title,full_time_position,year,avg(prevailing_wage) as average from h1b_final where full_time_position ='Y' and year='2011' and case_status in ('CERTIFIED','CERTIFIED-WITHDRAWN') group by job_title,full_time_position,year order by average desc;
